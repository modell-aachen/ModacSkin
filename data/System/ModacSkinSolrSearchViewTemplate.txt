%TMPL:DEF{"solr::format::list::topic"}%<!-- -->
<div class='solrSearchHit solrTopicHit'>
  <h3 style='background-image:url(%icon%)' title='%MAKETEXT{%type%}%'>
    [[%web%.%topic%]]
    <span class='foswikiGrayText'>
      %WEBLINK{"%web%"
        format="<a class='webLink' href='$url'>%MAKETEXT{"in [_1]" args="$title"}%</a>"
      }%
    </span>
  </h3>
  <div class='solrHilite'>
    <literal>%hilite%</literal>
  </div>
  <div class='solrRevision'>
    %date%,
    %IF{"istopic '%USERINFO{"%author%" format="$wikiusername"}%'"
      then="%USERINFO{"%author%" format="[[$wikiusername][$wikiname]]"}%"
      else="%USERINFO{"%author%" format="<nop>$wikiname"}%"
    }%
  </div>
</div>
<!-- -->%TMPL:END%
